<div id="app" class="p-5">
                <btn-count></btn-count>
                <br><br>
                <select-weekday></select-weekday>
            </div>
            
            <script>
                const app = Vue.createApp({})

                app.component('btn-count', {
                    data() {
                        return {
                            count: 0
                        }
                    },
                    template: `
                        <button class="btn btn-primary" @click="count += 1">
                            Click Count: {{count}}
                        </button>
                    `
                })

                app.component('select-weekday', {
                    data() {
                        return {
                            days: ['Sunday', 'Monday', '...', 'Saturday'],
                            selectedDay: ''
                        }
                    },
                    methods: {
                        showDay() {
                            alert(this.selectedDay)
                        }
                    },
                    template: `
                        <select class="form-control" @change="showDay" v-model="selectedDay">
                            <option v-for="d in days" :value="d">{{d}}</option>
                        </select>
                    `
                })

                app.mount('#app')
            </script>
