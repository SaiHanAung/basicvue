{{keyword}} 
            <div id="app" class="p-5">
                <form-search @click-button="keyword = $event"></form-search>
                <!-- click-button คือ custom event ที่ส่งมาจาก child -->
                <!-- $event คือค่าที่ส่งมาจาก child -->
                <span>{{keyword}}</span>
            </div>
            
            <script>
                const app = Vue.createApp({
                    data() {
                        return {
                            keyword: ''
                        }
                    },
                })

                app.component('form-search', {
                    template: `
                        <form>
                            <div class="input-group">
                                <input type="text" class="form-control" ref="textKw" placeholder="Search">
                                <button type="button" class="btn btn-primary" @click="$emit('click-button', this.$refs.textKw.value)">OK</button>
                            </div>
                        </form>
                    `
                    //สร้าง custom event ที่ชื่อ click-button
                    //พร้อมแนบค่าจากอินพุต text ไปด้วย
                })

                app.mount('#app')
            </script>
