            <div id="app" class="mx-auto" style="max-width: 300px;">
                <input type="text" class="form-control mb-1" v-model="sampleText">
                <span>ข้อมูลในอินพุต: {{sampleText}}</span><br>
                <button class="btn btn-primary mt-4" @click="btn_click">แก้ไขพร็อปเพอร์ตี้</button>
            </div>
            
            <script>
                Vue.createApp({
                    data() {
                        return { sampleText: 'Hello' }  //ค่าเริ่มต้น
                    },
                    methods: {
                        btn_click() {
                            this.sampleText = prompt('กำหนดค่า:') 
                        }
                    }
                }).mount('#app')
            </script>
