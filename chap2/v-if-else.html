สมาชิกเข้าสู่ระบบ
ท่านเข้าสู๋ระบบแล้ว
            <div class="text-center">
                <div id="app" class="m-4 d-inline-block text-center">
                    <form v-if="!signined" class="row g-2 m-auto p-3 bg-success bg-opacity-25" style="max-width: 300px;">
                        <h6 class="col-12 text-center mb-3 mb-sm-4">สมาชิกเข้าสู่ระบบ</h6>        
                        <div class="col-12">
                            <input type="text" class="form-control form-control-sm" placeholder="Login" ref="login">
                        </div>
                        <div class="col-12">
                            <input type="password" class="form-control form-control-sm" placeholder="Password" ref="password">
                        </div>
                        <div class="col-12 text-center pt-2">
                            <button class="btn btn-sm btn-primary me-3" @click="btnSignin_click">ตกลง</button>
                        </div> 
                    </form> 
                    <div v-else class="bg-info bg-opacity-25 p-4">
                        <h5>ท่านเข้าสู๋ระบบแล้ว</h5>
                        <button class="btn btn-sm btn-danger mt-4" @click="signined = false">ออกจากระบบ</button>
                    </div>
                </div>
            </div>
            
            <script>
                Vue.createApp({
                    data() {
                        return {
                            signined: false
                        }
                    },
                    methods: {
                        btnSignin_click() {
                            let lg = this.$refs.login.value.trim()
                            let pw = this.$refs.password.value.trim() 
                            if ( lg != '' && pw != '') {
                                this.signined = true
                            } else {
                                this.signined = false
                                alert('กรุณาใส่ Login และ Password')
                            }
                        }
                    }
                }).mount('#app')
            </script>
