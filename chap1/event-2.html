            <div id="app">
                <nav class="navbar navbar-light bg-primary bg-opacity-25 py-1 ps-3 pe-4">
                    <ul class="navbar-nav">
                        <li class="nav-item"><a class="nav-link" href="#">Home</a></li>
                    </ul>
                    <span class="position-relative">
                        <i class="bi-cart" style="color: blue; font-size: 24px;"></i>
                        <span class="position-absolute start-50 badge rounded-pill bg-danger">
                            {{ cart || '' }}    <!-- หรือ {{ (cart > 0) ? cart : '' }} -->
                        </span>
                    </span>
                </nav>  
                <div class="mt-4 text-center">
                    <div class="mb-2">คงเหลือ {{ remaining }}</div>
                    <div class="d-inline-block">
                        <span>จำนวนที่เลือก:</span>
                        <div class="input-group">
                            <button class="btn btn-outline-success" @click="dec">-</button> 
                            <span class="d-flex align-items-center px-3 border border-success">{{quantity}}</span>
                            <button class="btn btn-outline-success" @click="inc">+</button> 
                        </div>
                    </div>
                    <button class="btn btn-primary mt-3 d-block m-auto" @click="addCart">เพิ่มลงในรถเข็น</button>
                    <button class="btn btn-danger mt-3 d-block m-auto" @click="clearCart">ยกเลิกสินค้าในรถเข็น</button>
                </div>
            </div>
            
            <script>
                Vue.createApp({
                    data() {
                        return {
                            remaining: 5,       //คงเหลือ
                            quantity: 1,        //จำนวนที่เลือก (เพิ่ม-ลด)            
                            cart: 0,            //จำนวนในรถเข็น (Badge)
                        }
                    },
                    methods: {
                        dec() {
                            if (this.quantity > 2) {
                                this.quantity--
                            }
                        },
                        inc() {
                            if (this.quantity < this.remaining) {
                                this.quantity++
                            }                   
                        },
                        addCart() {
                            this.cart = this.quantity
                        },
                        clearCart() {
                            this.cart = 0
                        }
                    }
                }).mount('#app')
            </script>
